## E) CI 片段（签名/公证/健康门禁）
```yaml
# .github/workflows/release.yml（摘要）
name: Release Pipeline
on: 
  push: { tags: ['v*'] }
  workflow_dispatch: { inputs: { channel: { required: true, type: choice, options: [dev, staging, prod] } } }

jobs:
  build:
    strategy: { matrix: { os: [windows-latest, macos-latest] } }
    steps:
    - name: Build & Package
      run: npm run build && npm run package:${{ matrix.os }}
    - name: Sign (Windows)
      if: matrix.os == 'windows-latest'
      run: powershell -ExecutionPolicy Bypass -File scripts/release/windows-sign.ps1
      env: { WINDOWS_CERT_FILE: '${{ secrets.WIN_CERT }}', WINDOWS_CERT_PASSWORD: '${{ secrets.WIN_CERT_PASS }}' }
    - name: Notarize (macOS)
      if: matrix.os == 'macos-latest'
      run: node scripts/release/macos-notarize.mjs
      env: { APPLE_ID: '${{ secrets.APPLE_ID }}', APPLE_PASSWORD: '${{ secrets.APPLE_PASSWORD }}' }
    - name: Health Gate
      run: node scripts/release/health-gate-check.mjs --channel=${{ github.event.inputs.channel || 'prod' }}
      env: { SENTRY_ORG: '${SENTRY_ORG}', SENTRY_PROJECT: '${SENTRY_PROJECT}' }
    - name: Upload to CDN
      run: node scripts/release/upload-artifacts.mjs --target=${RELEASE_CDN}/${{ github.event.inputs.channel }}
```
